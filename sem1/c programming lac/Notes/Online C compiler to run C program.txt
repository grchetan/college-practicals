#include <stdio.h>

int main() {
    int a = 5;  // 5 = 0101 (in binary)
    int b = 9;  // 9 = 1001 (in binary)
    int result;

    // Bitwise AND Operator
    result = a & b;  // 0101 & 1001 = 0001 (binary) = 1 (decimal)
    printf("a & b = %d\n", result);

    // Bitwise OR Operator
    result = a | b;  // 0101 | 1001 = 1101 (binary) = 13 (decimal)
    printf("a | b = %d\n", result);

    // Bitwise XOR Operator
    result = a ^ b;  // 0101 ^ 1001 = 1100 (binary) = 12 (decimal)
    printf("a ^ b = %d\n", result);

    // Bitwise NOT Operator
    result = ~a;     // ~0101 = 1010 (binary) = -6 (decimal)
    printf("~a = %d\n", result);

    // Bitwise Left Shift Operator
    result = a << 1;  // 0101 << 1 = 1010 (binary) = 10 (decimal)
    printf("a << 1 = %d\n", result);

    // Bitwise Right Shift Operator
    result = b >> 1;  // 1001 >> 1 = 0100 (binary) = 4 (decimal)
    printf("b >> 1 = %d\n", result);

    return 0;
}
